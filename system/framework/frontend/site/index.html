m.var:imported = "__config.html-folder__/example.json";

CONFIG PRINT_R

        
            &m.var:print_r:config;

IMPORTED PRINT_R

        
            &m.var:print_r:imported;

GET PRINT_R

           &m.var:print_r:_GET;

GET JSON

           &m.var:json:_GET;

GET['if'] BOOLEAN

           &m.var:bool:_GET.if;
           &m.var:boolean:_GET.if;

GET['if'] INTEGER

           &m.var:int:_GET.if;
           &m.var:integer:_GET.if;

GET['if'] STRING

           &m.var:str:_GET.if;
           &m.var:string:_GET.if;

FOREACH USING GET

          
            __KEY__ -> &m.var:_GET.__KEY__;

IF

          
            _GET.if is false
          
          
            _GET.if is true

REPEAT

          
            __KEY__

INCLUDE

&m.include:__config.html-folder__/include.html;
